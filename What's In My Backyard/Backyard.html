/*

 1 Ear - > 2, 3, 4, 5

 4 Mirror - > 1
 5 Sky - > 1

import processing.sound.*;

SoundFile scream, musicBox, vinyl, bugs, scissors, jumpscare;

PImage ear;
PImage grass;
PImage bug;
PImage scissors1;
PImage scissors2;
PImage mirror;
PImage mirror2;
PImage grassBackyard;
PImage eyes;

float a;

int scene = 1;

int b1x = 370;  // Button 1 x-coordinate
int b1y = 310;  // Button 1 y-coordinate

int b2x = 750;  // Button 2 x-coordinate
int b2y = 310;  // Button 2 y-coordinate

int b3x = 50;  // Button 3 x-coordinate
int b3y = 310;  // Button 3 y-coordinate

int b4x = 400;  // Button 4 x-coordinate
int b4y = 50;  // Button 4 y-coordinate

int hx = 35;  // "Home" button x-coordinate
int hy = 550;  // "Home" button y-coordinate

int bdiameter = 50;  // Button diameter
int hdiameter = 30;  // "Home" button diameter

void setup() {
  size(800, 600);
  ear = loadImage("ear.png");
  grass = loadImage("grass.jpg");
  bug = loadImage("bug.png");
  scissors1 = loadImage("scissors1.png");
  scissors2 = loadImage("scissors2.png");
  mirror = loadImage("mirror.png");
  mirror2 = loadImage("mirror2.png");
  grassBackyard = loadImage("grassBackyard.png");
  eyes = loadImage("eyes.png");

  scream = new SoundFile(this, "scream.mp3");
  musicBox = new SoundFile(this, "musicBox.mp3");
  vinyl = new SoundFile(this, "vinyl.mp3");
  bugs = new SoundFile(this, "bugs.mp3");
  scissors = new SoundFile(this, "scissors.mp3");
  jumpscare = new SoundFile(this, "jumpscare.mp3");

  imageMode(CENTER);
  rectMode(CENTER);
  a = height;
}

void draw() {
  if (scene == 1) {
    drawScene1();
  } else if (scene == 2) {
    drawScene2();
  } else if (scene == 3) {
    drawScene3();
  } else if (scene == 4) {
    drawScene4();
  } else if (scene == 5) {
    drawScene5();
  }
}

void drawScene1() {
  if (!musicBox.isPlaying() &  !vinyl.isPlaying()) {
    musicBox.loop();
    vinyl.loop();
  }
  frameRate(10);
  tint(#99bfb4);
  image (grass, width/2, height/2);
  tint(0, 150);
  image (ear, width/2, height/2, random(200, 600), random(200, 500));
  tint(#5c877b);
  image (ear, width/2, height/2);
  noStroke();
  if (overCircle(b1x, b1y, bdiameter)) { // BUTTON 1
    fill(100, 0, 0, 85);
  } else {
    fill(0, 60);
  }
  circle(b1x, b1y, bdiameter);

  if (overCircle(b2x, b2y, bdiameter)) { // BUTTON 2
    fill(150, 0, 0, 85);
  } else {
    fill(255, 75);
  }
  circle(b2x, b2y, bdiameter);

  if (overCircle(b3x, b3y, bdiameter)) { // BUTTON 3
    fill(150, 0, 0, 85);
  } else {
    fill(255, 75);
  }
  circle(b3x, b3y, bdiameter);

  if (overCircle(b4x, b4y, bdiameter)) { // BUTTON 4
    fill(150, 0, 0, 85);
  } else {
    fill(255, 75);
  }
  circle(b4x, b4y, bdiameter);

  textSize(12);
  fill(255);
  text("Scene 1", 20, 30);
}

void drawBug() {
  image(bug, width-random(width), a, 75, 100);
  a = a - random(1, 6);
  rotate(random(50));
}

void drawScene2() {
  vinyl.pause();
  background(0);
  noStroke();
  if (!bugs.isPlaying()) {
    bugs.loop();
  }
  if (overCircle(hx, hy, hdiameter)) {
    fill(100, 0, 0);
    textSize(100);
    text("TAKE ME OUT!", hx+10, hy);
    if (!scream.isPlaying()) {
      scream.play();
    }
  } else {
    fill(100);
  }
  circle(hx, hy, hdiameter);
  fill(255);
  textSize(12);
  text("Scene 2", 20, 30);
  frameRate(5);
  rotate(random(50));
  tint(100);
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  drawBug();
  if (a < 0) {
    a = height;
  }
}

void drawScene3() {
  tint(#99bfb4);
  image(grass, width/2, height/2);
  if (overCircle(hx, hy, hdiameter)) {
    fill(100, 0, 0);
  } else {
    fill(100);
  }

  circle(hx, hy, hdiameter);
  fill(255);
  textSize(12);
  text("Scene 3", 20, 30);
  if (overRect(width/2, height/2, scissors1.width, scissors1.height)) {
    if (!scissors.isPlaying()) {
      scissors.play();
    }
    image(scissors2, width/2, height/2, 560*0.8, 380*0.8);
    image(scissors2, random(width), random(height), 560*0.5, 380*0.5);
    image(scissors1, random(width), random(height), 560*0.5, 380*0.5);
    image(scissors2, random(width), random(height), 560*0.1, 380*0.1);
    image(scissors1, random(width), random(height), 560*0.1, 380*0.1);
    image(scissors2, random(width), random(height), 560*0.7, 380*0.7);
    image(scissors1, random(width), random(height), 560*0.7, 380*0.7);
    image(scissors2, random(width), random(height), 560*0.3, 380*0.3);
    fill(100, 0, 0, 50);
    rect(width/2, height/2, width, height);
  } else {
    image(scissors1, width/2, height/2, 560*0.8, 380*0.8);
  }
}

void drawScene4() {
  tint(#99bfb4);
  image (grass, width/2, height/2);
  if (overCircle(hx, hy, hdiameter)) {
    fill(100, 0, 0);
  } else {
    fill(100);
  }
  circle(hx, hy, hdiameter);
  fill(255);
  textSize(12);
  text("Scene 4", 20, 30);
  image (mirror, width/2, height/2, 500, 500);
  if (overCircle(width/2+10, height/2-100, 100)) {

    if (mousePressed)
      image(mirror2, width/2 + random(-2, 2), height/2 + random(-2, 2), 500, 500);

    for (int i = 0; i < 10; i++) {
      noFill();
      strokeWeight(random(15));
      stroke(100, 0, 0, 75);
      circle(width/2+10, height/2-100, random(70, 800));
    }
  }
}

void drawScene5() {
  background(0);
  tint(#99bfb4);
  image (grassBackyard, width/2, height/2+50);
  if (overRect(width/2, height/2, scissors1.width, scissors1.height)) {
    image (eyes, width/2 + random(-2, 2), height/2 + random(-2, 2), 500, 500);
    image (grassBackyard, width/2, height/2+50);
    if (!jumpscare.isPlaying()) {
      jumpscare.play(1.5);
    }
  }
  if (overCircle(hx, hy, hdiameter)) {
    fill(100, 0, 0);
  } else {
    fill(100);
  }
  circle(hx, hy, hdiameter);
  fill(255);
  textSize(12);
  text("Scene 5", 20, 30);
}

void mousePressed() {
  if (scene == 1) {
    if (overCircle(b1x, b1y, bdiameter)) {
      scene = 2;
      println("go to scene 2");
    }
    if (overCircle(b2x, b2y, bdiameter)) {
      scene = 3;
      println("go to scene 3");
    }
    if (overCircle(b3x, b3y, bdiameter)) {
      scene = 4;
      println("go to scene 4");
    }
    if (overCircle(b4x, b4y, bdiameter)) {
      scene = 5;
      println("go to scene 5");
    }
  } else if (scene == 2) {
    if (overCircle(hx, hy, hdiameter)) {
      scene = 1;
      bugs.pause();
      if (!vinyl.isPlaying()) {
        musicBox.loop();
        vinyl.loop();
      }
      println("back to scene 1");
    }
  } else if (scene == 3) {
    if (overCircle(hx, hy, hdiameter)) {
      scene = 1;
      println("back to scene 1");
    }
  } else if (scene == 4) {
    if (overCircle(hx, hy, hdiameter)) {
      scene = 1;
      println("back to scene 1");
    }
  } else if (scene == 5) {
    if (overCircle(hx, hy, hdiameter)) {
      scene = 1;
      println("back to scene 1");
    }
  }
}

boolean overRect(float x, float y, float w, float h) {
  // If the (x,y) are inside the rect, return 'true'
  if (mouseX > x && mouseX < x+w &&
    mouseY > y && mouseY < y+h) {
    return true;
  } else {
    return false;
  }
}

boolean overCircle(float x, float y, float d) {
  // If the (x,y) are inside the circle, return 'true'
  if (dist(mouseX, mouseY, x, y) < d/2) {
    return true;
  } else {
    return false;
  }
}
